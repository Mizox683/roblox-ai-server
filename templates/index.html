📁 Server Script — ServerScriptService

Create a Script inside ServerScriptService and paste this. Replace PASTE_YOUR_RAI_KEY_HERE with your key.
local HttpService = game:GetService("HttpService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local API_KEY = "PASTE_YOUR_RAI_KEY_HERE"
local SERVER_URL = "https://roblox-ai-server-11uh.onrender.com/chat"
local AI_NAME = "Assistant"
local AI_PERSONALITY = "You are a helpful assistant inside a Roblox game. Be friendly, fun and keep responses short."
local DataStoreService = game:GetService("DataStoreService")
local chatStore = DataStoreService:GetDataStore("AIChatHistory_v2")

local remoteFolder = Instance.new("Folder")
remoteFolder.Name = "AIRemotes"
remoteFolder.Parent = ReplicatedStorage

local chatFunction = Instance.new("RemoteFunction")
chatFunction.Name = "AIChat"
chatFunction.Parent = remoteFolder

local saveFunction = Instance.new("RemoteFunction")
saveFunction.Name = "SaveHistory"
saveFunction.Parent = remoteFolder

local loadFunction = Instance.new("RemoteFunction")
loadFunction.Name = "LoadHistory"
loadFunction.Parent = remoteFolder

local clearFunction = Instance.new("RemoteFunction")
clearFunction.Name = "ClearHistory"
clearFunction.Parent = remoteFolder

chatFunction.OnServerInvoke = function(player, message)
    local success, result = pcall(function()
        local response = HttpService:RequestAsync({
            Url = SERVER_URL,
            Method = "POST",
            Headers = {
                ["Content-Type"] = "application/json",
                ["ngrok-skip-browser-warning"] = "true"
            },
            Body = HttpService:JSONEncode({
                api_key = API_KEY,
                player_name = player.Name,
                message = message,
                npc_name = AI_NAME,
                npc_personality = AI_PERSONALITY
            })
        })
        if response.Success then
            local data = HttpService:JSONDecode(response.Body)
            if data.status == "success" then
                local remaining = data.data.messages_remaining
                local reply = data.data.reply
                local lower = string.lower(message)
                if string.find(lower, "how many") or string.find(lower, "messages left") or string.find(lower, "remaining") then
                    return reply .. "\n\n💬 You have " .. tostring(remaining) .. " messages remaining!"
                end
                return reply
            elseif data.status == "limit_reached" then
                return "AI chat limit reached for this game!"
            end
        end
        return "Could not connect to AI server."
    end)
    return success and result or "Something went wrong."
end

saveFunction.OnServerInvoke = function(player, data)
    pcall(function()
        chatStore:SetAsync(tostring(player.UserId) .. "_data", data)
    end)
    return true
end

loadFunction.OnServerInvoke = function(player)
    local success, data = pcall(function()
        return chatStore:GetAsync(tostring(player.UserId) .. "_data")
    end)
    if success and data then return data end
    return {}
end

clearFunction.OnServerInvoke = function(player)
    local success, data = pcall(function()
        return chatStore:GetAsync(tostring(player.UserId) .. "_data")
    end)
    local cooldownEnd = 0
    local messageCount = 0
    if success and data then
        cooldownEnd = data.cooldown_end or 0
        messageCount = data.message_count or 0
    end
    pcall(function()
        chatStore:SetAsync(tostring(player.UserId) .. "_data", {
            messages = {},
            message_count = messageCount,
            cooldown_end = cooldownEnd
        })
    end)
    return true
end

local ReplicatedStorage = game:GetService("ReplicatedStorage")

local remoteFolder = ReplicatedStorage:WaitForChild("AIRemotes")

local cooldownEndTime = 0

local function getTimeRemaining()
    local remaining = cooldownEndTime - os.time()
    if remaining <= 0 then return "0 hours" end
    local hours = math.floor(remaining / 3600)
    local minutes = math.floor((remaining % 3600) / 60)
    if hours > 0 then return hours .. "h " .. minutes .. "m"
    else return minutes .. "m" end
end

local function updateCooldownDisplay()
    if isOnCooldown then
        cooldownLabel.Text = "⏳ Daily limit reached!\nCome back in " .. getTimeRemaining()
        cooldownOverlay.Visible = true
        textBox.TextEditable = false
        sendBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
        sendBtn.Text = "⏳"
    else
        cooldownOverlay.Visible = false
        textBox.TextEditable = true
        sendBtn.BackgroundColor3 = Color3.fromRGB(80, 100, 200)
        sendBtn.Text = "Send"
    end
end

local function updateCounter()
    counterLabel.Text = messageCount .. "/" .. MESSAGE_LIMIT
    if messageCount >= MESSAGE_LIMIT * 0.8 then
        counterLabel.TextColor3 = Color3.fromRGB(255, 150, 50)
    end
    if messageCount >= MESSAGE_LIMIT then
        counterLabel.TextColor3 = Color3.fromRGB(255, 80, 80)
    end
end

    local bubbleColor = isUser and Color3.fromRGB(60, 80, 180) or Color3.fromRGB(40, 40, 60)
    local align = isUser and Enum.TextXAlignment.Right or Enum.TextXAlignment.Left

    bubble.Size = UDim2.new(1, 0, 0, 0)
    bubble.AutomaticSize = Enum.AutomaticSize.Y
    bubble.BackgroundTransparency = 1
    bubble.Parent = scrollFrame
    local senderLabel = Instance.new("TextLabel")
    senderLabel.Size = UDim2.new(1, 0, 0, 16)
    senderLabel.BackgroundTransparency = 1
    senderLabel.Text = sender
    senderLabel.TextColor3 = Color3.fromRGB(150, 150, 200)
    senderLabel.TextSize = 11
    senderLabel.Font = Enum.Font.GothamBold
    senderLabel.TextXAlignment = align
    senderLabel.Parent = bubble
    local msgFrame = Instance.new("Frame")
    msgFrame.Size = UDim2.new(0.85, 0, 0, 0)
    msgFrame.AutomaticSize = Enum.AutomaticSize.Y
    msgFrame.Position = isUser and UDim2.new(0.15, 0, 0, 18) or UDim2.new(0, 0, 0, 18)

    msgPadding.Parent = msgFrame
    task.wait(0.05)
    scrollFrame.CanvasPosition = Vector2.new(0, scrollFrame.AbsoluteCanvasSize.Y)
end

local function saveHistory()
    saveFunction:InvokeServer({
        messages = chatHistory,
        message_count = messageCount,
        cooldown_end = cooldownEndTime
    })
end

local function loadHistory()
    local data = loadFunction:InvokeServer()
    if data and data.messages and #data.messages > 0 then
        chatHistory = data.messages
        messageCount = data.message_count or #data.messages
        cooldownEndTime = data.cooldown_end or 0
        if cooldownEndTime > 0 and os.time() < cooldownEndTime then
            isOnCooldown = true
        else
            isOnCooldown = false
            cooldownEndTime = 0
        end
        for _, msg in ipairs(chatHistory) do
            addBubble(msg.sender, msg.text, msg.isUser)
        end
        updateCounter()
        updateCooldownDisplay()
    end
end

local function activateCooldown()
    isOnCooldown = true
    cooldownEndTime = os.time() + (COOLDOWN_HOURS * 3600)
    saveHistory()
    updateCooldownDisplay()
    addBubble("⚠️ System", "You've reached the 50 message daily limit! Come back in 24 hours.", false)
end

local isSending = false
local function sendMessage()
    if isSending then return end
    if isOnCooldown then updateCooldownDisplay() return end
    if messageCount >= MESSAGE_LIMIT then activateCooldown() return end
    local message = textBox.Text
    if message == "" then return end
    isSending = true
    sendBtn.Text = "..."
    sendBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
    textBox.Text = ""
    addBubble("You", message, true)
    messageCount = messageCount + 1
    table.insert(chatHistory, {sender = "You", text = message, isUser = true})
    updateCounter()
    if messageCount >= MESSAGE_LIMIT then
        isSending = false
        sendBtn.Text = "Send"
        sendBtn.BackgroundColor3 = Color3.fromRGB(80, 100, 200)
        activateCooldown()
        return
    end
    addBubble("🤖 Assistant", "Thinking...", false)
    local reply = chatFunction:InvokeServer(message)
    local bubbles = scrollFrame:GetChildren()
    for i = #bubbles, 1, -1 do
        if bubbles[i]:IsA("Frame") then bubbles[i]:Destroy() break end
    end
    addBubble("🤖 Assistant", reply, false)
    table.insert(chatHistory, {sender = "🤖 Assistant", text = reply, isUser = false})
    saveHistory()
    sendBtn.Text = "Send"
    sendBtn.BackgroundColor3 = Color3.fromRGB(80, 100, 200)
    isSending = false
end

clearBtn.MouseButton1Click:Connect(function()
    for _, child in pairs(scrollFrame:GetChildren()) do
        if child:IsA("Frame") then child:Destroy() end
    end
    chatHistory = {}
    clearFunction:InvokeServer()
    if isOnCooldown then
        updateCooldownDisplay()
        addBubble("⚠️ System", "Chat cleared but you're still on cooldown! Come back in " .. getTimeRemaining(), false)
    else
        messageCount = 0
        updateCounter()
        addBubble("🤖 Assistant", "Chat cleared! How can I help you?", false)
    end
end)

sendBtn.MouseButton1Click:Connect(sendMessage)
textBox.FocusLost:Connect(function(enterPressed)
    if enterPressed then sendMessage() end
end)

task.spawn(function()
    while true do
        task.wait(60)
        if isOnCooldown then
            if os.time() >= cooldownEndTime then
                isOnCooldown = false
                cooldownEndTime = 0
                messageCount = 0
                updateCounter()
                updateCooldownDisplay()
                addBubble("✅ System", "Your daily limit has reset! You can chat again.", false)
                saveHistory()
            else
                updateCooldownDisplay()
            end
        end
    end
end)

task.wait(1)
loadHistory()

if #chatHistory == 0 and not isOnCooldown then
    addBubble("🤖 Assistant", "Hey " .. player.Name .. "! I'm your AI assistant. Ask me anything! (50 messages per day)", false)
end
